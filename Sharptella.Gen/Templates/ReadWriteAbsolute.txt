    private bool $OP$_ReadWriteAbsolute()
    {
        switch (m_CurrentOpCodeCycle)
        {
            case 1:
                m_SavedValue0 = m_Bus.Read(m_Registers.PC);
                ++m_Registers.PC;
                return false;
            case 2:
                m_SavedValue1 = m_Bus.Read(m_Registers.PC);
                ++m_Registers.PC;
                Trace($"$OP$ ${m_SavedValue0:X2}{m_SavedValue1:X2}");
                return false;
            case 3:
                m_SavedValue2 = m_Bus.Read(GetAbsolute(m_SavedValue0, m_SavedValue1));
                return false;
            case 4:
                m_Bus.Write(GetAbsolute(m_SavedValue0, m_SavedValue1), m_SavedValue2);
                return false;
            default:
                m_Bus.Write(GetAbsolute(m_SavedValue0, m_SavedValue1), $OP$(m_SavedValue2));
                return true;
        }
    }
